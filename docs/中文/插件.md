## 插件开发文档

---

### 概述

桌宠引擎支持通过插件扩展功能。每个插件是一个独立的文件夹，包含一个 `enter.py` 入口文件。

插件可以添加右键菜单项，菜单项点击后执行对应的逻辑。插件也可以设置启动时自动执行。

---

### 目录结构

```
data/
└── 桌宠目录/
    ├── config.json          # 桌宠配置
    ├── plugin/              # 插件目录
    │   └── [插件名]/         # 单个插件文件夹
    │       └── enter.py     # 插件入口文件（必须）
    └── res/                 # 桌宠资源
```

---

### 启用插件

在桌宠的 `config.json` 中，通过 `plugin` 字段声明需要加载的插件：

```json
{
  "name": "Example",
  "version": "1.0.0",
  "author": "YourName",
  "acc": [0, 1.2],
  "fri": {
    "top": 2,
    "bottom": 3,
    "left": 0,
    "right": 0
  },
  "ela": {
    "top": 0,
    "bottom": 5,
    "left": 10,
    "right": 10
  },
  "plugin": ["message", "reminder"]
}
```

`plugin` 数组中的每个字符串对应 `plugin/` 目录下的一个插件文件夹名称。

### 插件入口文件（enter.py）

每个插件必须包含 `enter.py`，并定义以下内容：

#### 1. `pluginName`（必须）

插件的显示名称，会出现在右键菜单中作为子菜单的标题。

```python
pluginName = "弹窗消息"
```

#### 2. `menu` 字典（必须）

定义右键菜单项。每个键为菜单显示名称，值为一个对象实例。当用户点击菜单项时，主程序会调用该对象的 `enter` 方法。

```python
menu = {
    "消息": Information(),
    "问题": Question(),
    "警告": Warning(),
    "错误": Error()
}
```

#### 3. `enter` 方法（必须）

每个菜单项类必须实现 `enter` 方法，当用户点击菜单项时调用：

```python
def enter(self, image, mainTimer, physicsTimer, state, window):
    """
    菜单项被点击时的回调方法

    参数:
        image: QMovie 对象，用于控制桌宠动画
        mainTimer: QTimer 对象，主循环定时器（20ms）
        physicsTimer: QTimer 对象，物理循环定时器（20ms）
        state: dict，桌宠状态字典，可读写
        window: QWidget 对象，主窗口
    """
    pass
```

#### 4. `__autoStart__` 属性（可选）

在类中定义 `__autoStart__` 属性，控制是否在程序启动时自动执行 `enter` 方法：

```python
class Information(Template):
    def __init__(self):
        super().__init__()
        self.__autoStart__ = True  # 启动时自动执行
```

### 完整示例：弹窗消息插件

```python
from PySide6 import QtCore, QtGui, QtWidgets

# 基础模板类
class Template:
    def __init__(self):
        self.__autoStart__ = False

    def enter(self, image, mainTimer, physicsTimer, state, window):
        self.__image__ = image
        self.__mainTimer__ = mainTimer
        self.__physicsTimer__ = physicsTimer
        self.__state__ = state
        self.__window__ = window

    def loadMovie(self, path: str):
        """切换桌宠动画"""
        if self.__image__.fileName() != path:
            self.__image__.setFileName(path)
            self.__image__.jumpToFrame(0)
        self.__image__.start()


# 具体功能类
class Information(Template):
    def enter(self, image, mainTimer, physicsTimer, state, window):
        super().enter(image, mainTimer, physicsTimer, state, window)
        QtWidgets.QMessageBox.information(window, "消息", "这是一条提示消息")


class Question(Template):
    def enter(self, image, mainTimer, physicsTimer, state, window):
        super().enter(image, mainTimer, physicsTimer, state, window)
        QtWidgets.QMessageBox.question(window, "问题", "确定要执行此操作吗？")


class Warning(Template):
    def enter(self, image, mainTimer, physicsTimer, state, window):
        super().enter(image, mainTimer, physicsTimer, state, window)
        QtWidgets.QMessageBox.warning(window, "警告", "这是一个警告信息")


class Error(Template):
    def enter(self, image, mainTimer, physicsTimer, state, window):
        super().enter(image, mainTimer, physicsTimer, state, window)
        QtWidgets.QMessageBox.critical(window, "错误", "发生了一个错误")


# 插件元数据（必须定义）
pluginName = "弹窗消息"

menu = {
    "消息": Information(),
    "问题": Question(),
    "警告": Warning(),
    "错误": Error()
}
```

### 带 __autoStart__ 的插件示例

```python
from PySide6 import QtCore, QtGui, QtWidgets
import random

class Template:
    def __init__(self):
        self.__autoStart__ = False

    def enter(self, image, mainTimer, physicsTimer, state, window):
        self.__image__ = image
        self.__mainTimer__ = mainTimer
        self.__physicsTimer__ = physicsTimer
        self.__state__ = state
        self.__window__ = window


class Reminder(Template):
    def __init__(self):
        super().__init__()
        self.__autoStart__ = True  # 启动时自动执行

    def enter(self, image, mainTimer, physicsTimer, state, window):
        super().enter(image, mainTimer, physicsTimer, state, window)
        print("提醒插件已加载！")
        
        # 启动定时器，每30秒弹窗提醒
        timer = QtCore.QTimer()
        timer.timeout.connect(self.show_reminder)
        timer.start(30000)
        self.__state__["reminder_timer"] = timer

    def show_reminder(self):
        messages = ["该休息一下了！", "起来走动走动！", "喝口水吧！"]
        QtWidgets.QMessageBox.information(
            self.__window__, 
            "提醒", 
            random.choice(messages)
        )


class StopReminder(Template):
    def enter(self, image, mainTimer, physicsTimer, state, window):
        super().enter(image, mainTimer, physicsTimer, state, window)
        if "reminder_timer" in state:
            state["reminder_timer"].stop()
            state["reminder_timer"].deleteLater()
            del state["reminder_timer"]
            print("提醒已关闭")


# 插件元数据（必须定义）
pluginName = "定时提醒"

menu = {
    "开启提醒": Reminder(),
    "关闭提醒": StopReminder()
}
```

### 带状态更新的插件示例

如果需要在每一帧执行逻辑，可以将函数注册到 `state["update"]` 中：

```python
from PySide6 import QtCore, QtGui, QtWidgets

class Template:
    def __init__(self):
        self.__autoStart__ = False

    def enter(self, image, mainTimer, physicsTimer, state, window):
        self.__image__ = image
        self.__mainTimer__ = mainTimer
        self.__physicsTimer__ = physicsTimer
        self.__state__ = state
        self.__window__ = window


class AutoMove(Template):
    def __init__(self):
        super().__init__()
        self.__autoStart__ = True

    def enter(self, image, mainTimer, physicsTimer, state, window):
        super().enter(image, mainTimer, physicsTimer, state, window)
        
        def update():
            # 每帧执行：让桌宠左右摆动
            import math
            time = QtCore.QDateTime.currentMSecsSinceEpoch() / 1000
            state["motion"][0] = math.sin(time) * 2
            state["motion"][1] = -1
        
        state["update"]["auto_move"] = update


pluginName = "自动运动"

menu = {
    "开启自动运动": AutoMove()
}
```
